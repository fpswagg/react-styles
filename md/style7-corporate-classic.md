class Security {

    // Sanitisation des entrÃ©es
    public static function sanitizeInput($input) {
        if (is_array($input)) {
            return array_map([self::class, 'sanitizeInput'], $input);
        }

        return htmlspecialchars(strip_tags(trim($input)), ENT_QUOTES, 'UTF-8');
    }

    // Validation CSRF
    public static function generateCSRFToken() {
        if (empty($_SESSION['csrf_token'])) {
            $_SESSION['csrf_token'] = bin2hex(random_bytes(32));
        }
        return $_SESSION['csrf_token'];
    }

    public static function validateCSRFToken($token) {
        return isset($_SESSION['csrf_token']) && hash_equals($_SESSION['csrf_token'], $token);
    }

    // Authentification
    public static function authenticateUser($email, $password) {
        // VÃ©rification des credentials
        $stmt = Database::prepare("
            SELECT employee_id, password_hash, role, active
            FROM employees
            WHERE email = ? AND active = 1
        ");

        $stmt->execute([$email]);
        $user = $stmt->fetch();

        if ($user && password_verify($password, $user['password_hash'])) {
            // CrÃ©er session sÃ©curisÃ©e
            session_regenerate_id(true);
            $_SESSION['user_id'] = $user['employee_id'];
            $_SESSION['user_role'] = $user['role'];
            $_SESSION['last_activity'] = time();

            return $user;
        }

        return false;
    }

    // Autorisation
    public static function checkPermission($permission, $userRole = null) {
        $role = $userRole ?? ($_SESSION['user_role'] ?? 'guest');

        $permissions = [
            'admin' => ['read', 'write', 'delete', 'admin'],
            'manager' => ['read', 'write', 'approve'],
            'employee' => ['read', 'write'],
            'guest' => []
        ];

        return in_array($permission, $permissions[$role] ?? []);
    }

    // Audit logging

            INSERT INTO audit_log (user_id, action, resource, resource_id, ip_address, user_agent, created_at)

            $_SESSION['user_id'] ?? null,
            $action,
            $resource,
            $resourceId,
            $_SERVER['REMOTE_ADDR'],
            $_SERVER['HTTP_USER_AGENT']
        ]);
